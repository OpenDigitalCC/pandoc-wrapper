%% pipeline-preamble.tex
%%
%% Portable preamble carrying every LaTeX dependency the pandoc-wrapper Lua
%% filter's output relies on - tables (datatables), boxes (::: divs) and charts
%% (pie/bar) - plus the handful of generic typographic settings the pipeline
%% assumes. It is deliberately template-INDEPENDENT: it contains no Eisvogel or
%% KOMA-specific commands, so any base template that pulls it in (via \input or
%% by header-includes injection) can render the full feature set.
%%
%% This is the "shim" layer: the common ground between a stock base template
%% (e.g. Eisvogel or the MVP) and the brand layer.
%%
%% Idempotent: safe to \input more than once (a wrapper template and the MVP may
%% both pull it in). The second load stops here.
\ifdefined\pipelinePreambleLoaded\endinput\fi
\def\pipelinePreambleLoaded{}

% xcolor must see the table option before anything else loads it (hyperref etc).
\PassOptionsToPackage{table}{xcolor}

% --- Tables (datatable feature: longtable + coloured rows + row spans) ---
\usepackage{longtable}
\usepackage{booktabs}
\usepackage{array}
\usepackage{multirow}
\usepackage{tabularx}
\usepackage{colortbl}

% --- Boxes (::: divs rendered to tcolorbox by the filter) ---
\usepackage[most]{tcolorbox}
\usepackage{needspace}
\usepackage{wrapfig}
\usepackage{fontawesome5}
% marginbox divs emit \marginnote plus \checkoddpage/\ifoddpage (the filter's
% output) so the note can lean into the outer margin. Load both packages here so
% the feature works under every base template, not just those that happen to pull
% them in. Without these, ::: marginbox fails with "Undefined control sequence
% \marginnote" (then \checkoddpage) - exit 43.
\usepackage{marginnote}
\usepackage{changepage}\strictpagecheck % \checkoddpage needs strict checking

% --- Charts (piechart / barchart rendered to TikZ/pgf) ---
\usepackage{pgfplots}
\usepackage{pgf-pie}
\pgfplotsset{compat=1.18}

% --- Endnotes (footnote citations collected into an end-matter section) ---
% The package and its look live here (template-independent); a base template
% activates it per-brand via footnotes-as-endnotes / endnotes-* variables.
\usepackage{endnotes}
\makeatletter
\renewcommand\enoteformat{%
  \raggedright\leftskip=1.8em\parindent=-1.8em%
  \leavevmode\llap{\makeenmark}%
}
\renewcommand\enotesize{\small}
\makeatother

% --- Generic typography / layout the pipeline assumes (brands may override) ---
\usepackage{microtype}
\usepackage{parskip}
\usepackage{caption}
\captionsetup{labelfont=bf}  % brands override with \captionsetup, never a re-\usepackage
\setkeys{Gin}{width=1\textwidth,keepaspectratio}
\renewcommand{\tabcolsep}{2mm}
\renewcommand{\arraystretch}{1.1}
\renewcommand{\arrayrulewidth}{1mm}
\hyphenpenalty=5000
\tolerance=1000
