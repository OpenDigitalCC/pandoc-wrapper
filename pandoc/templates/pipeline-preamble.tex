%% pipeline-preamble.tex
%%
%% Portable preamble carrying every LaTeX dependency the pandoc-wrapper Lua
%% filter's output relies on - tables (datatables), boxes (::: divs) and charts
%% (pie/bar) - plus the handful of generic typographic settings the pipeline
%% assumes. It is deliberately template-INDEPENDENT: it contains no Eisvogel or
%% KOMA-specific commands, so any base template that pulls it in (via \input or
%% by header-includes injection) can render the full feature set.
%%
%% This is the "shim" layer: the common ground between a stock base template
%% (e.g. Eisvogel or the MVP) and the brand layer.
%%
%% Idempotent: safe to \input more than once (a wrapper template and the MVP may
%% both pull it in). The second load stops here.
\ifdefined\pipelinePreambleLoaded\endinput\fi
\def\pipelinePreambleLoaded{}

% xcolor must see the table option before anything else loads it (hyperref etc).
\PassOptionsToPackage{table}{xcolor}

\usepackage{longtable}

\usepackage{multirow}

\usepackage{colortbl}

% --- Boxes (::: divs rendered to tcolorbox by the filter) ---
\usepackage[most]{tcolorbox}
\usepackage{needspace}
\usepackage{wrapfig}
\usepackage{fontawesome5}
% marginbox divs emit \marginnote plus \checkoddpage/\ifoddpage (the filter's
% output) so the note can lean into the outer margin. Load both packages here so
% the feature works under every base template, not just those that happen to pull
% them in. Without these, ::: marginbox fails with "Undefined control sequence
% \marginnote" (then \checkoddpage) - exit 43.
\usepackage{marginnote}
\usepackage{changepage}\strictpagecheck % \checkoddpage needs strict checking
% Widen margin notes to fill the outer margin (the marginnote default ~2cm looks
% cramped in brands' wide-margin report layouts). \pwMarginNoteWidth is called at
% the point of use (in the marginbox / margin-chart output - see
% document-filters.lua), NOT in a preamble hook: KOMA-Script recalculates the
% type area - and \marginparwidth - in its own \AtBeginDocument hook that runs
% after any we could register, so a preamble value gets clobbered. Computed from
% the live geometry so it adapts to whatever margins the brand set; floored at
% 1cm so a narrow-margin document can never get a zero/negative width.
% \checkoddpage (under \strictpagecheck) only reports the TRUE page parity on a
% second LaTeX run - it labels the page on one run and reads that label back on
% the next. Pandoc runs xelatex once unless the log requests a rerun, so on a
% single pass every margin note guesses its side and can lean to the wrong
% (inner) edge - or, on a verso page, fail to swap to the outer margin at all.
% Every margin placement (marginbox and margin charts) calls \pwMarginNoteWidth,
% so arm a one-shot rerun there: write a "settled" marker into the .aux at
% begin-document (it is read back on the NEXT run only) and, if it is still
% undefined at end-document while a margin note was used, emit a warning pandoc
% recognises as a rerun request. This forces exactly the second pass that
% \checkoddpage needs, and only when the document actually has margin notes.
\makeatletter
\newif\ifpw@marginused
\newcommand{\pwMarginNoteWidth}{%
  \global\pw@marginusedtrue
  \setlength{\marginparsep}{4mm}%
  \setlength{\marginparwidth}{\dimexpr\paperwidth-\oddsidemargin-1in-\textwidth-\marginparsep-6mm\relax}%
  \ifdim\marginparwidth<1cm\setlength{\marginparwidth}{1cm}\fi}
\AtBeginDocument{\immediate\write\@mainaux{%
  \string\global\string\let\string\pw@margin@settled\string\@empty}}
\AtEndDocument{%
  \ifpw@marginused
    \@ifundefined{pw@margin@settled}{%
      \@latex@warning@no@line{Rerun to get margin-note page parity right}}{}%
  \fi}
\makeatother

% ===================================================================
% Code blocks (skylighting / default highlighting path)
% -------------------------------------------------------------------
% Present fenced code as a panel: a light fill, a brand-coloured rule down
% the left edge and a gently rounded corner, so code reads as a distinct
% object rather than indented prose. Brands tune two colours:
%
%   code-background-color : the panel fill          -> colour "code-background"
%   code-accent-color     : left rule + keywords    -> colour "code-accent"
%
% document-filters.lua resolves a brand-colour NAME in either field to a hex
% value and emits the matching \definecolor; the \providecolor lines below are
% the neutral fallback used when a brand sets neither.
%
% Line numbers are on by default (the filter adds .numberLines to every plain
% code block; opt a single block out with the .nonumber class). fancyvrb draws
% them in the margin, so they are never part of the copyable text.
% ===================================================================
% fvextra provides line wrapping (breaklines) for highlighted blocks. The
% eisvogel template loads it; the other base templates use bare fancyvrb, where
% a long line would overflow the panel. Loading it here (it pulls in fancyvrb,
% and a second load under eisvogel is a harmless no-op) makes wrapping uniform.
\usepackage{fvextra}
% accsupp lets us give the line numbers empty "ActualText" so they stay visible
% but are skipped by PDF text extraction - i.e. they are not picked up when the
% reader copies the code (genuinely outside the copy region, not just visually).
\usepackage{accsupp}
\providecolor{code-background}{HTML}{F7F7F7}
\providecolor{code-accent}{HTML}{435489}
\providecolor{code-linenum}{HTML}{B0B0B0}
\providecolor{code-string}{HTML}{00999A}
\providecolor{code-comment}{HTML}{8E8E8E}
\providecolor{code-function}{HTML}{8250DF}

% The panel: a breakable tcolorbox so long listings span pages. Registered at
% begin-document so it overrides pandoc's own (empty) Shaded environment from
% $highlighting-macros$, whatever order the base template loads things in.
% (tcolorbox + the breakable/skins libraries are already loaded, [most], above.)
\AtBeginDocument{%
  \ifcsname Shaded\endcsname
    \renewenvironment{Shaded}{%
      \begin{tcolorbox}[
        breakable, enhanced,
        colback=code-background, colframe=code-accent,
        boxrule=0pt, leftrule=2.2pt, arc=2pt, outer arc=2pt,
        boxsep=0pt, left=24pt, right=7pt, top=6pt, bottom=6pt,
        enlarge top by=2pt, enlarge bottom by=3pt,
        before skip=8pt, after skip=10pt
      ]%
    }{%
      \end{tcolorbox}%
    }%
  \fi
}

% Line-number look: small, muted, set a little away from the code. Also turn on
% line wrapping for every highlighted block: the eisvogel template sets
% breaklines on its own Highlighting env, but the stock one the other templates
% use does not, so a long line would overflow the panel. Setting it as a global
% fancyvrb default fixes that uniformly without touching each template's
% font-size handling.
\AtBeginDocument{%
  \ifdefined\theFancyVerbLine
    \renewcommand{\theFancyVerbLine}{%
      \BeginAccSupp{ActualText={}}%
      \color{code-linenum}\scriptsize\sffamily\arabic{FancyVerbLine}%
      \EndAccSupp{}}%
    \fvset{numbersep=13pt,breaklines=true,breaksymbolleft={},breakautoindent=false}%
  \fi
}

% Token colours: keep skylighting's defaults for most tokens, but pull the
% keyword family onto the brand accent and settle strings/comments/functions
% onto a calm, legible palette. Applied here directly (not in a begin-document
% hook, which mangles the #1 parameter): every base template that \input's this
% file does so AFTER pandoc's $highlighting-macros$, so the \...Tok commands
% already exist. Guarded so a document with no code is untouched.
\ifdefined\KeywordTok
  \renewcommand{\KeywordTok}[1]{\textcolor{code-accent}{\textbf{#1}}}
  \renewcommand{\ControlFlowTok}[1]{\textcolor{code-accent}{\textbf{#1}}}
  \renewcommand{\StringTok}[1]{\textcolor{code-string}{#1}}
  \renewcommand{\CharTok}[1]{\textcolor{code-string}{#1}}
  \renewcommand{\SpecialStringTok}[1]{\textcolor{code-string}{#1}}
  \renewcommand{\VerbatimStringTok}[1]{\textcolor{code-string}{#1}}
  \renewcommand{\CommentTok}[1]{\textcolor{code-comment}{\textit{#1}}}
  \renewcommand{\CommentVarTok}[1]{\textcolor{code-comment}{\textit{#1}}}
  \renewcommand{\DocumentationTok}[1]{\textcolor{code-comment}{\textit{#1}}}
  \renewcommand{\FunctionTok}[1]{\textcolor{code-function}{#1}}
\fi
